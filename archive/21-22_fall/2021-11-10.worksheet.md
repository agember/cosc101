```python

```

### Example


```python
def func3():
    num = int(input('Enter a number: '))
    return num

def func2():
    input_num = func3()
    return input_num * 2

def func1():
    doubled_num = func2()
    return doubled_num * 2

def main():
    quadrupled_num = func1()
    print("Quadrupled:", quadrupled_num)

main()
```
