```python
def read_data(filename):
    '''Read data from a file'''
    infile = open(filename, 'r')
    data = infile.read()
    infile.close()
    return data

def main():
    data = read_data('seasons.txt')
    print(data)
    data = read_data('does-not-exist.txt')
    print(data)
    
main()
```

```python
def read_data(filename):
    '''Read data from a file'''
    try:
        infile = open(filename, 'r')
        data = infile.read()
        infile.close()
        return data
    except:
        return None

def main():
    data = read_data('seasons.txt')
    print(data)
    data = read_data('does-not-exist.txt')
    print(data)
    
main()
```
